// ============================================ // INITIALIZE - FIXED // ============================================ document.addEventListener('DOMContentLoaded', function() { renderNavigation(); const user = getCurrentUser(); const currentPath = window.location.pathname; const isLoginPage = currentPath === '/login/'; const isLogoutPage = currentPath === '/logout/'; const isFinancePage = currentPath.includes('/dashboard/finance/'); // ✅ IMPORTANT: Skip all session checks on login page if (isLoginPage) { console.log('On login page - skipping all session checks'); return; } if (isLogoutPage) { console.log('On logout page - skipping session check'); return; } // Only verify session if user exists in localStorage and not on login page if (user && user.id) { console.log('Verifying session for user:', user.username); fetch('/api/auth/current_user/', { credentials: 'include', headers: { 'X-Requested-With': 'XMLHttpRequest' } }) .then(response => { console.log('Session check response status:', response.status); if (response.status === 404) { return user; } if (response.status === 401 || response.status === 403) { throw new Error('Session invalid'); } return response.json(); }) .then(data => { if (data && data.id) { console.log('Session valid for:', data.username); localStorage.setItem('user', JSON.stringify(data)); renderNavigation(); } else if (!isFinancePage) { console.log('Session invalid, redirecting to login'); localStorage.removeItem('user'); renderNavigation(); window.location.href = '/login/'; } }) .catch(error => { console.log('Session check error:', error.message); if (error.message === 'Session invalid' && !isFinancePage) { localStorage.removeItem('user'); renderNavigation(); window.location.href = '/login/'; } }); } else if (!isLoginPage) { console.log('No user found, redirecting to login'); window.location.href = '/login/'; } // Subscribe form const subscribeForm = document.querySelector('.subscribe'); if (subscribeForm) { subscribeForm.addEventListener('submit', function(e) { e.preventDefault(); const email = this.querySelector('#subscribe-email'); if (email && email.value) { showNotification('Thank you for subscribing!', 'success'); email.value = ''; } else { showNotification('Please enter a valid email address.', 'warning'); } }); } });
